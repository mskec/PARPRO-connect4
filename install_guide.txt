# Installing MPI

apt-get install python-dev
apt-get install python-pip
apt-get install git

apt-get install mpich2
pip install mpi4py

# Generating ssh key
ssh-keygen -t rsa

# Deploying app
git clone git@github.com:mskec/PARPRO-Connect4.git


# Running
mpirun -np 4 python src/connect4.py

cat /proc/cpuinfo
